362         public void initializeLevels(String leftJoins, EntityManager em) {
363             // Prevent n+1 queries for each level
364             int i = 0;
365             do {
366                 List<Level> subList = levels.subList(i, Math.min(levels.size(), i + DB.MAX_LIST_EXPRESSIONS));
367                 em.createQuery("from Level l "
368                         + leftJoins + " "
369                         + "where l in (:levels) ")
370                         .setParameter("levels", subList)
371                         .getResultList();
372                 i += subList.size();
373             } while (i < levels.size());
374         }
375 
376         public void initializeApplicationLayers(String leftJoins, EntityManager em) {
377             if (!getApplicationLayers().isEmpty()) {
378                 // Prevent n+1 queries for each ApplicationLayer
379                 int i = 0;
380                 do {
381                     List<ApplicationLayer> subList = applicationLayers.subList(i, Math.min(applicationLayers.size(), i + DB.MAX_LIST_EXPRESSIONS));
382                     em.createQuery("from ApplicationLayer al "
383                             + leftJoins + " "
384                             + "where al in (:alayers) ")
385                             .setParameter("alayers", subList)
386                             .getResultList();
387                     i += subList.size();
388                 } while (i < applicationLayers.size());
389             }
390         }
391     }

400     public TreeCache loadTreeCache(EntityManager em) {
401         if (treeCache == null) {
402 
403             treeCache = new TreeCache();
404 
405             // Retrieve level tree structure in single query
406             treeCache.levels = em.createNamedQuery("getLevelTree", Level.class)
407                     .setParameter("rootId", root.getId())
408                     .getResultList();
409 
410             treeCache.childrenByParent = new HashMap<>();
411             treeCache.applicationLayers = new ArrayList<>();
412 
413             for (Level l : treeCache.levels) {
414                 treeCache.applicationLayers.addAll(l.getLayers());
415 
416                 if (l.getParent() != null) {
417                     List<Level> parentChildren = treeCache.childrenByParent.computeIfAbsent(l.getParent(), k -> new ArrayList<>());
418                     parentChildren.add(l);
419                 }
420             }
421 
422         }
423         return treeCache;
424     }

430     /**
431      * keep a cached copy of our mashup status.
432      */
433     @Transient
434     public Boolean isMashup_cached;
435 
436     /**
437      * slow method -especially on Oracle- because it will need all keys and
438      * Oracle lazy loading sucks so clob's get pulled in as well.
439      *
440      * @return {@code true} if we are a mashup
441      * @see #isMashup(org.hibernate.Session)
442      */
443     public Boolean isMashup() {
444         if (this.getDetails().containsKey(Application.DETAIL_IS_MASHUP)) {
445             String mashupValue = this.getDetails().get(Application.DETAIL_IS_MASHUP).getValue();
446             this.isMashup_cached = Boolean.valueOf(mashupValue);
447         } else {
448             this.isMashup_cached = Boolean.FALSE;
449         }
450         return this.isMashup_cached;
451     }
452 
453     /**
454      * fast access to determine if we are mashup.
455      *
456      * @param sess the hibernate session
457      * @return {@code true} if we are a mashup, this may be a cached value
458      *
459      * @see #isMashup()
460      */
461     public Boolean isMashup(Session sess) {
462         if (this.isMashup_cached == null) {
463             this.isMashup_cached = Boolean.FALSE;
464             List<Map<String, Object>> items = sess.createSQLQuery("select d.value from application_details d where d.details_key = '" + Application.DETAIL_IS_MASHUP + "' and d.application =  :appId")
465                     .setParameter("appId", this.getId())
466                     .setResultTransformer(ApplicationDetailsValueTransformer.INSTANCE)
467                     .list();
468             if (!items.isEmpty()) {
469                 String mashupValue = items.get(0).get("value").toString();
470                 this.isMashup_cached = Boolean.valueOf(mashupValue);
471             }
472         }
473         return this.isMashup_cached;
474     }

478         List<ApplicationLayer> appLayers = treeCache.applicationLayers;
479         for (ApplicationLayer appLayer : appLayers) {
480             if (appLayer.getService().equals(oldLayer.getService()) && appLayer.getLayerName().equals(oldLayer.getLayerName())) {
481                 return appLayer;
482             }
483         }
484         return null;
485     }

506     @Transient
507     public Map<String, Long> idMap;
508 
509     public void processCopyMap() {
510         if (originalToCopy == null) {
511             log.debug("postPersist(): not a copy");
512             return;
513         }
514         idMap = getIdMap();
515 
516         originalToCopy = null;
517 
518         log.debug("Updating component configs");
519         for (ConfiguredComponent comp : components) {
520             if (comp.getConfig() == null) {
521                 continue;
522             }
523             log.debug(String.format("Checking component class %s, name %s", comp.getClassName(), comp.getName()));
524             boolean changed = false;
525             try {
526                 JSONObject cfg = new JSONObject(comp.getConfig());
527                 if (cfg.has("layers")) {
528                     JSONArray layers = cfg.getJSONArray("layers");
529                     for (int i = 0; i < layers.length(); i++) {
530                         Long newId = idMap.get(ApplicationLayer.class + "_" + layers.getInt(i));
531                         if (newId != null) {
532                             log.debug(String.format("Index %d: new id for application layer %d is %d", i, layers.getInt(i), newId));
533                             layers.put(i, newId.longValue());
534                         } else {
535                             log.debug(String.format("Index %d: old id %d was not a valid application layer in original", i, layers.getInt(i)));
536                             layers.put(i, -1);
537                         }
538                         changed = true;
539                     }
540                 }
541                 if (cfg.has("levels")) {
542                     JSONArray levels = cfg.getJSONArray("levels");
543                     for (int i = 0; i < levels.length(); i++) {
544                         Long newId = idMap.get(Level.class + "_" + levels.getInt(i));
545                         if (newId != null) {
546                             log.debug(String.format("Index %d: new id for level %d is %d", i, levels.getInt(i), newId));
547                             levels.put(i, newId.longValue());
548                         } else {
549                             log.debug(String.format("Index %d: old id %d was not a valid level in original", i, levels.getInt(i)));
550                             levels.put(i, -1);
551                         }
552                         changed = true;
553                     }
554                 }
555                 if (changed) {
556                     log.debug("Old config: " + comp.getConfig());
557                     comp.setConfig(cfg.toString());
558                     log.debug("New config: " + comp.getConfig());
559                 }
560 
561             } catch (Exception ex) {
562                 log.error(String.format("Cannot update persistent object id's "
563                         + "in component config on application copy, "
564                         + "copied application=%s, component class=%s, component name=%s",
565                         getNameWithVersion(),
566                         comp.getClassName(),
567                         comp.getName()), ex);
568             }
569         }
570     }

572     private Map<String, Long> getIdMap() {
573         Map<String, Long> idMap = new HashMap<>();
574         for (Object e : originalToCopy.entrySet()) {
575             Map.Entry<Object, Object> entry = (Map.Entry<Object, Object>) e;
576             Object original = entry.getKey();
577             Object copy = entry.getValue();
578             log.debug(String.format("postPersist(): original=%s, copy=%s", original, copy));
579             if (original instanceof Level) {
580                 Level oL = (Level) original;
581                 Level cL = (Level) copy;
582                 idMap.put(original.getClass() + "_" + oL.getId(), cL.getId());
583             } else if (original instanceof ApplicationLayer) {
584                 ApplicationLayer oAl = (ApplicationLayer) original;
585                 ApplicationLayer cAl = (ApplicationLayer) copy;
586                 idMap.put(original.getClass() + "_" + oAl.getId(), cAl.getId());
587             }
588         }
589         return idMap;
590     }

593     public void removeOldProperties() {
594         // In previous versions maxHeight and maxWidth where assigned to details directly
595         // Now these settings are saved in globalLayout. We are removing these settings from
596         // details (when present) to migrate from old layout to new layout
597         this.details.remove("maxWidth");
598         this.details.remove("maxHeight");
599     }

601     public void setGlobalLayout(String globalLayout) {
602         this.details.put("globalLayout", new ClobElement(globalLayout));
603     }

605     public JSONObject getGlobalLayout() throws JSONException {
606         JSONObject globalLayout = new JSONObject();
607         if (this.getDetails().containsKey("globalLayout")) {
608             globalLayout = new JSONObject(this.getDetails().get("globalLayout").getValue());
609         }
610         // Legacy properties
611         if (!globalLayout.has("maxWidth") && this.getDetails().containsKey("maxWidth")) {
612             globalLayout.put("maxWidth", this.getDetails().get("maxWidth").getValue());
613         }
614         if (!globalLayout.has("maxHeight") && this.getDetails().containsKey("maxHeight")) {
615             globalLayout.put("maxHeight", this.getDetails().get("maxHeight").getValue());
616         }
617         return globalLayout;
618     }
