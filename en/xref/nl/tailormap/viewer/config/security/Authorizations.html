18  /**
19   * Utility class for authorization checking. Because of the inheritence of 
20   * authorizations in tree structures, caches are used for efficient 
21   * authorization checks.
22   * <p>
23   * There are authorizations on these objects:
24   * <p>
25   * <b>Geo services registry:</b>
26   * <ul>
27   * <li>Categories: currently disabled, not accessible via user interface</li>
28   * <li>Layers of GeoServices: with inheritence, read/write authorizations</li>
29   * </ul>
30   * <p>
31   * <b>Application:</b>
32   * <ul>
33   * <li>Levels: with inheritence, read authorization only</li>
34   * <li>ApplicationLayers of Levels: inherits from Level <i>and</i> the referenced Layer, read/write authorizations</li>
35   * <li>ConfiguredComponents</li>
36   * </ul>
37   * <p>
38   * Authorizations are based on role names which are Group names.
39   * <p>
40   * <b>Inheritence and authorization rules:</b>
41   * <ol>
42   * <li>The empty set of authorized role names (EVERYBODY) means
43   * everybody is authorized.</li>
44   * <li>A non-empty set which is not equal to NOBODY means only users who
45   * are members of the named Groups in the set are authorized</li>
46   * <li>NOBODY is the set containing a single value which is null.</li>
47   * <li>Inheritence processing starts at the root with everybody authorized.</li>
48   * <li>When inheriting authorizations, the set of authorized roles can only be
49   * reduced and not expanded. When a parent is authorized for roles A and B and
50   * the set of authorized roles for a child is B and C, only B is authorized. When
51   * the child is only authorized for role C, nobody will be authorized for the child.</li>
52   * <li>Because of the default access for everybody, denying editing for example is done
53   * by only authorizing write access to a group which has no members.</li>
54   * </ol>
55   * <b>Caching</b>
56   * <ul>
57   * <li>Layer authorizations are cached and invalidated when authorizations are 
58   * changed for a GeoService by updating the GeoService.authorizationsModified timestamp.</li>
59   * <li>Level and ApplicationLayer authorizations are cached and invalidated when
60   * authorizations on a Level or ApplicationLayer are changed by updating the 
61   * Application.authorizationsModified timestamp.</li>
62   * <li>Because ApplicationLayer authorizations inherit from Layer authorizations,
63   * Application caches are invalidated when any authorization on a Layer changes. 
64   * </ul>
65   * Although when authorizations are changed for an object only the cached object
66   * and its inheriting objects could be refreshed in the cache, the entire set of
67   * cached Layers of a GeoService or cached Levels and ApplicationLayers of an 
68   * Application are invalidated because:
69   * <ol>
70   * <li>The viewer-admin webapp is isolated from the viewer webapp so it cannot 
71   * access its application-scope cache without extra cross-webapp communication.</li>
72   * <li>The viewer-admin webapp communicates with the viewer webapp to refresh
73   * its cache by updating the above mentioned authorizationChanged database columns.</li>
74   * <li>The cache for an Application is invalidated when the maximum authorization 
75   * changed date (for all GeoServices) column is later than the Application authorization 
76   * changed date.</li>
77   * <li>These database columns are the simplest method to refresh the cache, and no
78   * partial updates are required to maintain performance because authorizations are
79   * modified infrequently.</li>
80   * <li>Applying inherited authorizations is optimized by loading a tree in a single
81   * query using hierarchical queries.</li>
82   * </ol>
83   * <p>
84   * Group membership for a User is not cached and read from the database for
85   * the first authorization check in a Transaction. So authorization changes by
86   * an administrator should always be directly applied except authorization checks
87   * which use HttpServletRequest.isUserInRole(), which is cached by the servlet
88   * container. Currently only viewer-admin checks roles this way.
89   * @author Matthijs Laan
90   */
91  public class Authorizations {
92      public static final String AUTHORIZATIONS_KEY = "authorizations";
93  
94      private static final Log log = LogFactory.getLog(Authorizations.class);
95  
96      public static boolean isFormAuthorized(Form f, Set<String> userRoles, EntityManager em) {
97          Set<String> formRoles = f.getReaders();
98          return !Collections.disjoint(userRoles, formRoles) || formRoles.isEmpty();
99      }

101 
102     public static boolean isUserExpired(User u) {
103         if(u == null){
104             return false;
105         }
106         Date today;
107         Date expire;
108         try {
109             DateFormat formatter = new SimpleDateFormat("dd-MM-yyyy");
110             expire = formatter.parse(u.getDetails().getOrDefault("expiry_date",formatter.format(new Date())));
111             today = formatter.parse(formatter.format(new Date()));
112         } catch (ParseException e){
113             log.error("Error parsing expiry_date for user: " + u.getUsername(), e);
114             return  true;
115         }
116         return today.after(expire);
117     }
