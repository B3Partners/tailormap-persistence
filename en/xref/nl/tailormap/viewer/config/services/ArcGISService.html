19  import nl.tailormap.viewer.config.ClobElement;
20  import org.json.JSONObject;
21  
22  import javax.persistence.DiscriminatorValue;
23  import javax.persistence.Entity;
24  import javax.persistence.Transient;
25  import java.util.List;
26  import java.util.Map;
27  import java.util.Set;
28  import java.util.SortedMap;
29  
30  /**
31   *
32   * @author Matthijs Laan
33   */
34  @Entity
35  @DiscriminatorValue(ArcGISService.PROTOCOL)
36  public class ArcGISService extends GeoService{
37  
38      public static final String PROTOCOL = "arcgis";
39  
40      /** Parameter to avoid the call to /ArcGIS/rest/services?f=json to determine
41       * the version (10 or 9). Some sites have this URL hidden but the service
42       * itself is available. String with "9" or "10", null or any other value
43       * means get it from /ArcGIS/rest/services?f=json.
44       */
45      public static final String PARAM_ASSUME_VERSION = "assumeVersion";
46  
47      /** GeoService.details map key for ArcGIS currentVersion property */
48      public static final String DETAIL_CURRENT_VERSION = "arcgis_currentVersion";
49  
50      /** GeoService.details map key to save assume version to pass on to datastore */
51      public static final String DETAIL_ASSUME_VERSION = "arcgis_assumeVersion";
52  
53      /** Layer.details map key for ArcGIS type property */
54      public static final String DETAIL_TYPE = "arcgis_type";
55      /** Layer.details map key for ArcGIS description property */
56      public static final String DETAIL_DESCRIPTION = "arcgis_description";
57      /** Layer.details map key for ArcGIS geometryType property */
58      public static final String DETAIL_GEOMETRY_TYPE = "arcgis_geometryType";
59      /** Layer.details map key for ArcGIS capabilities property */
60      public static final String DETAIL_CAPABILITIES = "arcgis_capabilities";
61      /** Layer.details map key for ArcGIS defaultVisibility property */
62      public static final String DETAIL_DEFAULT_VISIBILITY = "arcgis_defaultVisibility";
63      /** Layer.details map key for ArcGIS definitionExpression property */
64      public static final String DETAIL_DEFINITION_EXPRESSION = "arcgis_definitionExpression";
65  
66      public static final String TOPLAYER_ID = "-1";
67  
68      // Layer types are not specified in the ArcGIS API reference, so these are guesses.
69      // See {Layer#virtual}
70      // Group layers are thus virtual layers. Sometimes ArcGIS even has layers
71      // without a type...
72      public static final Set<String> NON_VIRTUAL_LAYER_TYPES = Set.of(new String[]{
73              "Feature Layer",
74              "Raster Layer",
75              "Annotation Layer" // not sure about this one...
76      });
77  
78  
79      @Transient
80      public JSONObject serviceInfo;
81      @Transient
82      public String currentVersion;
83      @Transient
84      public int currentVersionMajor;
85      @Transient
86      public SortedMap<String,Layer> layersById;
87      @Transient
88      public Map<String,List<String>> childrenByLayerId;
89  
90      public String getCurrentVersion() {
91          ClobElement ce = getDetails().get(DETAIL_CURRENT_VERSION);
92          String cv = ce != null ? ce.getValue() : null;
93  
94          if(cv == null && getTopLayer() != null) {
95              // get it from the topLayer, was saved there before GeoService.details
96              // was added
97              ce = getTopLayer().getDetails().get(DETAIL_CURRENT_VERSION);
98              cv = ce != null ? ce.getValue() : null;
99  
100             // try the first actual layer where may have been saved in version < 4.1
101             if(cv == null && !getTopLayer().getChildren().isEmpty()) {
102                 ce = getTopLayer().getChildren().get(0).getDetails().get(DETAIL_CURRENT_VERSION);
103                 cv = ce != null ? ce.getValue() : null;
104             }
105         }
106         return cv;
107     }
