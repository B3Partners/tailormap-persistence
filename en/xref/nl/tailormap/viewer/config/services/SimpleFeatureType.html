154     public boolean update(SimpleFeatureType update) {
155         if(!getTypeName().equals(update.getTypeName())) {
156             throw new IllegalArgumentException("Cannot update feature type with properties from feature type with different type name!");
157         }        
158 
159         description = update.description;
160         writeable = update.writeable;
161         geometryAttribute = update.geometryAttribute;
162         primaryKeyAttribute = update.primaryKeyAttribute;
163         
164         boolean changed = false;
165         
166         // Retain user set aliases for attributes
167         
168         // Does not work correctly for Arc* feature sources which set attribute
169         // title in alias... Needs other field to differentiate user set title
170         
171         Map<String,String> aliasesByAttributeName = new HashMap<>();
172         for(AttributeDescriptor ad: attributes) {
173             if(StringUtils.isNotBlank(ad.getAlias())) {
174                 aliasesByAttributeName.put(ad.getName(), ad.getAlias());
175             }
176         }
177         
178         //loop over oude attributes
179         // voor iedere oude attr kijk of er een attib ib de update.attributes zit
180         //   zo ja kijk of type gelijk is
181         //      als type niet gelijk dan oude attr verwijderen en vervangen door nieuwe, evt met alias kopieren
182         // zo niet dan toevoegen nieuw attr aan oude set
183         //   loop over nieuwe attributen om te kijken of er oude verwijderd moeten worden
184         // todo: Het is handiger om deze check op basis van 2 hashmaps uittevoeren
185         if(!attributes.equals(update.attributes)) {
186             changed = true;
187             for(int i = 0; i < attributes.size();i++){
188                 boolean notFound = true;
189                 
190                 for(AttributeDescriptor newAttribute: update.attributes){
191                     
192                     if(attributes.get(i).getName().equals(newAttribute.getName())){
193                         notFound = false;
194                         AttributeDescriptor oldAttr  = attributes.get(i);
195                         if(Objects.equals(oldAttr.getType(), newAttribute.getType())){ // NOPMD
196                             // ! expression didnt work(???) so dummy if-else (else is only used)
197                         }else{
198                             attributes.remove(i);
199                             attributes.add(i, newAttribute);  
200                         }
201                         break;
202                     }
203                 }
204                 if(notFound){
205                     attributes.remove(i);
206                 }
207             }
208             
209             //nieuwe attributen worden hier toegevoegd aan de oude attributen lijst
210             for(int i = 0; i < update.attributes.size();i++){
211                 boolean notFound = true;
212                 
213                 for(AttributeDescriptor oldAttribute: attributes){
214                     if(update.attributes.get(i).getName().equals(oldAttribute.getName())){
215                         notFound = false;
216                         break; 
217                     }
218                    
219                 }
220                 if(notFound){
221                     attributes.add(update.attributes.get(i));
222                 }
223             }
224         }
225         
226         //update.attributes ID = NULL so the attributes list is getting NULL aswell
227 
228         for(AttributeDescriptor ad: attributes) {
229             String alias = aliasesByAttributeName.get(ad.getName());
230             if(alias != null) {
231                 ad.setAlias(alias);
232             }            
233         }
234         return changed;
235     }
236     
237     public AttributeDescriptor getAttribute(String attributeName) {
238         for(AttributeDescriptor ad: attributes) {
239             if(ad.getName().equals(attributeName)) {
240                 return ad;
241             }
242         }
243         return null;
244     }

246     public JSONObject toJSONObject() throws JSONException {
247         JSONObject o = new JSONObject();
248         o.put("id", id);
249         o.put("typeName", typeName);
250         o.put("writeable", writeable);
251         o.put("geometryAttribute", geometryAttribute);
252         o.put("primaryKeyAttribute", primaryKeyAttribute);
253 
254         JSONArray atts = new JSONArray();
255         o.put("attributes", atts);
256         for(AttributeDescriptor a: attributes) {
257             JSONObject ja = new JSONObject();
258             ja.put("id", a.getId());
259             ja.put("name", a.getName());
260             ja.put("alias", a.getAlias());
261             ja.put("type", a.getType());
262             atts.put(ja);
263         }
264         return o;
265     }
