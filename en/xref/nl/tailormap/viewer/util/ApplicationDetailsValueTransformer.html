4   package nl.tailormap.viewer.util;
5   
6   import org.apache.commons.collections.map.CaseInsensitiveMap;
7   import org.apache.commons.logging.Log;
8   import org.apache.commons.logging.LogFactory;
9   import org.hibernate.transform.BasicTransformerAdapter;
10  
11  import java.sql.Clob;
12  import java.sql.SQLException;
13  import java.util.Map;
14  
15  /**
16   * Transform the clob in application details to a string. Avoid using with very
17   * large values.
18   *
19   * @author Mark Prins
20   */
21  public class ApplicationDetailsValueTransformer extends BasicTransformerAdapter {
22  
23      private static final Log LOG = LogFactory.getLog(ApplicationDetailsValueTransformer.class);
24      public final static ApplicationDetailsValueTransformer INSTANCE;
25  
26      static {
27          INSTANCE = new ApplicationDetailsValueTransformer();
28      }
29  
30      private ApplicationDetailsValueTransformer() {
31  
32      }
33  
34      /**
35       * create a CaseInsensitiveMap with key/values.
36       * <p>
37       * Tuples are the elements making up each "row" of the query result. The
38       * contract here is to transform these elements into the final row.
39       *
40       * @param tuple The result elements
41       * @param aliases The result aliases ("parallel" array to tuple)
42       * @return The transformed row.
43       */
44      @Override
45      public Object transformTuple(Object[] tuple, String[] aliases) {
46          Map<String, Object> map = new CaseInsensitiveMap();
47          for (int i = 0; i < aliases.length; i++) {
48              Object t = tuple[i];
49              if (t instanceof Clob) {
50                  Clob c = (Clob) tuple[i];
51                  try {
52                      t = c.getSubString(1, (int) c.length());
53                  } catch (SQLException e) {
54                      LOG.error("Error transforming data tuple of " + aliases[i], e);
55                  }
56              }
57              map.put(aliases[i], t);
58          }
59          return map;
60      }
61  }
