21  public class StartLevelTest extends TestUtil {

23      private static final Log log = LogFactory.getLog(StartLevelTest.class);
24  
25      @Test
26      public void persistLevel() {
27          StartLevel sl = new StartLevel();
28          sl.setSelectedIndex(16);
29          persistEntityTest(sl, StartLevel.class);
30  
31          entityManager.refresh(sl);
32          StartLevel test = entityManager.find(StartLevel.class, sl.getId());
33          assertNotNull(test);
34  
35          assertEquals(new Integer(16), test.getSelectedIndex());
36          assertEquals(6, entityManager.createQuery("FROM Level").getResultList().size());
37      }
38  
39      @Test
40      public void deleteStartLevel() throws URISyntaxException, SQLException, IOException {
41          Application app = entityManager.find(Application.class, applicationId);
42  
43          Level level = entityManager.find(Level.class, 5L);
44  
45          StartLevel sl = new StartLevel();
46          sl.setLevel(level);
47          sl.setApplication(app);
48          sl.setSelectedIndex(16);
49          persistEntityTest(sl, StartLevel.class);
50  
51          Level levelExists = entityManager.find(Level.class, 5L);
52          Application appExists = entityManager.find(Application.class, applicationId);
53  
54          assertNotNull(levelExists);
55          assertNotNull(appExists);
56          assertEquals(6, entityManager.createQuery("FROM Level").getResultList().size());
57      }
