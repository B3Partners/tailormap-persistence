6   import org.apache.commons.lang3.StringUtils;
7   
8   import java.util.ArrayList;
9   import java.util.List;
10  import java.util.Map;
11  import java.util.Set;
12  
13  /**
14   *
15   * @author Matthijs Laan
16   */
17  public class RemoveEmptyMapValuesUtil {
18  
19      /**
20       * IMPORTANT TO CALL THIS BEFORE SAVING / UPDATING AN ENTITY WITH A DETAILS
21       * MAP. Hibernate fails to remove map entries with a null value when calling
22       * clear() on a map on Oracle. This leads to duplicate key violations when
23       * updating layer details.
24       * <p>
25       * {@code @PreUpdate}/{@code @PrePersist} nor {@code @EntityListeners} do
26       * not work consistenly with cascaded objects in Hibernate, so call this
27       * manually!
28       *
29       * @param map the map to clean up
30       */
31      public static void removeEmptyMapValues(Map map) {
32          List keysToRemove = new ArrayList();
33          for(Map.Entry entry: (Set<Map.Entry>)map.entrySet()) {
34              if(entry.getValue() == null 
35                  || (entry.getValue() instanceof String && StringUtils.isEmpty((String)entry.getValue()))
36                  || (entry.getValue() instanceof ClobElement && StringUtils.isEmpty(((ClobElement)entry.getValue()).getValue()))) {
37                  keysToRemove.add(entry.getKey());
38              }
39          }
40          for(Object key: keysToRemove) {
41              map.remove(key);
42          }
43      }
44  }
