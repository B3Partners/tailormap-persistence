189     public String getPath() {
190         Level l = this;
191         
192         String s = "";
193         do {
194             s = l.getName() + (s.length() == 0 ? "" : "/" + s);
195             l = l.getParent();
196         } while(l != null);
197         
198         return s;            
199     }
200     
201     public Level getParentInSubtree(ApplicationLayer appLayer){
202         for(ApplicationLayer al: layers) {
203             if(al.equals(appLayer)) {
204                 return this;
205             }
206         }
207         for(Level child: children) {
208             Level parent = child.getParentInSubtree(appLayer);
209             if(parent!=null) {
210                 return parent;
211             }
212         }
213         return null;
214     }
215     
216     public Level getParentInSubtree(Level level){
217         for(Level child: children) {
218             if(child.equals(level)) {
219                 return this;
220             }
221             Level parent = child.getParentInSubtree(level);
222             if(parent !=null) {
223                 return parent;
224             }
225         }        
226         return null;
227     }

229     public boolean containsLayerInSubtree(ApplicationLayer appLayer) {
230         return this.getParentInSubtree(appLayer)!=null;
231     }
232 
233     public boolean containsLevelInSubtree(Level level) {
234         return this.getParentInSubtree(level)!=null;
235     }

237     public boolean hasLayerInSubtree(){
238         if(!layers.isEmpty()){
239             return true;
240         }else{
241             for (Level level : children) {
242                 if(level.hasLayerInSubtree()){
243                     return true;
244                 }
245             }
246         }
247         return false;
248     }
249 
250     public boolean isInSubtreeOf(Level level) {
251         Level parentLevel = parent;
252         do {
253             if(parentLevel.equals(level)) {
254                 return true;
255             }
256             parentLevel = parentLevel.getParent();
257         } while(parentLevel != null);
258         return false;
259     }

261     public void processForMashup(Application mashup, Application motherApp) throws Exception{
262         for (Level child : children) {
263             child.processForMashup(mashup, motherApp);
264         }
265         for (ApplicationLayer layer : layers) {
266             layer.processStartLayers(mashup, layer, motherApp);
267         }
268         processStartLevels(mashup, this, motherApp);
269     }
270     
271     public void processForWorkversion(Application workversion, Application base) throws Exception{
272         for (Level child : children) {
273             child.processForWorkversion(workversion, base);
274         }
275         for (ApplicationLayer layer : layers) {
276             layer.processStartLayers(workversion, (ApplicationLayer)workversion.originalToCopy.get(layer), base);
277         }
278         processStartLevels(workversion, (Level)workversion.originalToCopy.get(this), base);
279     }
280     
281     private void processStartLevels(Application app, Level original, Application copyFrom) throws Exception{
282        StartLevel sl = original.getStartLevels().get(copyFrom);
283        if(sl != null){
284            this.getStartLevels().put(app, sl.deepCopy(app, this));
285        }else if (Objects.equals(app.getId(), copyFrom.getId())){
286             List<StartLevel> sls = new ArrayList<>(original.startLevels.values());
287            for (StartLevel value : sls) {
288                this.getStartLevels().put(app, value.deepCopy(app, this));
289            }
290         }
291     }

293     public Level deepCopy(Level parent, Map originalToCopy, Application app, boolean processStartMap) throws Exception {
294         Level copy = (Level)BeanUtils.cloneBean(this);
295         originalToCopy.put(this, copy);
296         copy.setId(null);
297         copy.setParent(parent);
298         
299         copy.setChildren(new ArrayList<>());
300         for(Level child: children) {
301             copy.getChildren().add(child.deepCopy(copy, originalToCopy, app,processStartMap));
302         }
303         
304         copy.setLayers(new ArrayList<>());
305         for(ApplicationLayer appLayer: layers) {
306             copy.getLayers().add(appLayer.deepCopy(originalToCopy, app,processStartMap));
307         }
308         
309         copy.setStartLevels(new HashMap<>());
310         if(processStartMap){
311             copy.processStartLevels(app, this, app);
312         }
313         
314         // do not clone documents, only the list
315         copy.setDocuments(new ArrayList<>(documents));
316         
317         copy.setReaders(new HashSet<>(readers));
318         
319         copy.setInfo(info);
320         
321         copy.setUrl(url);
322         
323         return copy;
324     }
