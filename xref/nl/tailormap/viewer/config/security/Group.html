4   package nl.tailormap.viewer.config.security;
5   
6   import org.hibernate.annotations.Type;
7   
8   import javax.persistence.Entity;
9   import javax.persistence.Id;
10  import javax.persistence.Lob;
11  import javax.persistence.ManyToMany;
12  import javax.persistence.Table;
13  import java.util.Arrays;
14  import java.util.Collections;
15  import java.util.HashSet;
16  import java.util.List;
17  import java.util.Set;
18  
19  /**
20   *
21   * @author Matthijs Laan
22   */
23  @Entity
24  @Table(name="group_")
25  public class Group {
26      public static final String ADMIN = "Admin";
27      public static final String REGISTRY_ADMIN = "RegistryAdmin";
28      public static final String USER_ADMIN = "UserAdmin";
29      public static final String APPLICATION_ADMIN = "ApplicationAdmin";
30      public static final String SERVICE_ADMIN = "ServiceAdmin";
31      public static final String EXTENDED_USER = "ExtendedUser";
32      
33      public static final List<String> DEFAULT_GROUPS = Collections.unmodifiableList(Arrays.asList(
34              ADMIN, 
35              REGISTRY_ADMIN, 
36              USER_ADMIN, 
37              APPLICATION_ADMIN, 
38              EXTENDED_USER, 
39              SERVICE_ADMIN));
40      
41      @Id
42      private String name;
43  
44      @Lob
45      @Type(type = "org.hibernate.type.TextType")
46      private String description;
47  
48      @ManyToMany(mappedBy="groups")
49      private Set<User> members = new HashSet<User>();
50  
51      public String getDescription() {
52          return description;
53      }
54  
55      public void setDescription(String description) {
56          this.description = description;
57      }
58  
59      public String getName() {
60          return name;
61      }
62  
63      public void setName(String name) {
64          this.name = name;
65      }
66  
67      public Set<User> getMembers() {
68          return members;
69      }
70  
71      public void setMembers(Set<User> members) {
72          this.members = members;
73      }
74  }
