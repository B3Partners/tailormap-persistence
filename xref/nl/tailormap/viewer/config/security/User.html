29  @Entity
30  @Table(name="user_")
31  public class User implements Principal{
32      // See edituser.jsp
33      public static final String DETAIL_EMAIL = "email";
34      
35      public static final int MIN_PASSWORD_LENGTH = 8;
36  
37      private static final String DIGEST_ALGORITM = "SHA-1";
38      private static final String DIGEST_CHARSET = "UTF-8";
39  
40      @Id
41      private String username;
42  
43      private String password;
44  
45      @ManyToMany
46      @JoinTable(name="user_groups", joinColumns=@JoinColumn(name="username"), inverseJoinColumns=@JoinColumn(name="group_"))
47      private Set<Group> groups = new HashSet<Group>();
48  
49      @ElementCollection
50      @JoinTable(joinColumns=@JoinColumn(name="username"))
51      private Map<String,String> details = new HashMap<String,String>();
52      
53      
54      @ElementCollection
55      @Column(name="ipaddress", length = 45)
56      @CollectionTable(joinColumns = @JoinColumn(name="user_"))
57      private Set<String> ips = new HashSet<String>();
58      
59      @Transient
60      private boolean authenticatedByIp = false;
61  
62      public void changePassword(String password) throws NoSuchAlgorithmException, UnsupportedEncodingException {
63  
64          MessageDigest md = (MessageDigest) MessageDigest.getInstance(DIGEST_ALGORITM);
65          md.update(password.getBytes(DIGEST_CHARSET));
66          byte[] digest = md.digest();
67  
68          /* Converteer byte array naar hex-weergave */
69          StringBuilder sb = new StringBuilder(digest.length*2);
70          for(int i = 0; i < digest.length; i++) {
71              sb.append(Integer.toHexString(digest[i] >> 4 & 0xf)); /* and mask met 0xf nodig door sign-extenden van bytes... */
72              sb.append(Integer.toHexString(digest[i] & 0xf));
73          }
74          setPassword(sb.toString());
75      }
76      
77      public String getPassword() {
78          return password;
79      }
80  
81      public void setPassword(String password) {
82          this.password = password;
83      }
84  
85      public Map<String, String> getDetails() {
86          return details;
87      }
88  
89      public void setDetails(Map<String, String> details) {
90          this.details = details;
91      }
92  
93      public Set<Group> getGroups() {
94          return groups;
95      }
96  
97      public void setGroups(Set<Group> groups) {
98          this.groups = groups;
99      }

117     public boolean checkRole(String role){
118         for (Group group : groups) {
119             if(group.getName().equals(role)){
120                 return true;
121             }
122         }
123         return false;
124     }
125 
126     public boolean isAuthenticatedByIp() {
127         return authenticatedByIp;
128     }
129 
130     public void setAuthenticatedByIp(boolean authenticatedByIp) {
131         this.authenticatedByIp = authenticatedByIp;
132     }
133 
134     @Override

136         return username;
137     }
