33  /**
34   *
35   * @author Matthijs Laan
36   */
37  @Entity
38  public class AttributeDescriptor {
39      public static final String TYPE_STRING = "string";
40      public static final String TYPE_DOUBLE = "double";
41      public static final String TYPE_INTEGER = "integer";
42      public static final String TYPE_BOOLEAN = "boolean";
43      public static final String TYPE_GEOMETRY = "geometry";    
44      public static final String TYPE_DATE = "date";
45      public static final String TYPE_TIMESTAMP = "timestamp";
46      public static final String TYPE_GEOMETRY_POINT = "point";
47      public static final String TYPE_GEOMETRY_MPOINT = "multipoint";
48      public static final String TYPE_GEOMETRY_LINESTRING = "linestring";
49      public static final String TYPE_GEOMETRY_MLINESTRING = "multilinestring";
50      public static final String TYPE_GEOMETRY_POLYGON = "polygon";
51      public static final String TYPE_GEOMETRY_MPOLYGON = "multipolygon";
52      
53      public static final Set<String> GEOMETRY_TYPES = Collections.unmodifiableSet(new HashSet(Arrays.asList(new String[] {
54          TYPE_GEOMETRY,
55          TYPE_GEOMETRY_POINT,
56          TYPE_GEOMETRY_MPOINT,
57          TYPE_GEOMETRY_LINESTRING,
58          TYPE_GEOMETRY_MLINESTRING,
59          TYPE_GEOMETRY_POLYGON,
60          TYPE_GEOMETRY_MPOLYGON
61      })));
62  
63      @Id
64      @GeneratedValue(strategy = GenerationType.IDENTITY)
65      private Long id;
66  
67      @Basic(optional=false)
68      private String name;
69  
70      @Column(name="name_alias")
71      private String alias;
72  
73      private String type;
74      
75      /**
76       * Returns the ExtJS
77       * <a href="https://docs.sencha.com/extjs/5.1/5.1.0-apidocs/#!/api/Ext.data.field.Field">field</a>
78       * type for this attribute.
79       *
80       * @return a string representing the field type (one of
81       * auto,string,int,number,boolean,date)
82       */
83      public String getExtJSType() {
84          // default to "auto" field type
85          String ext_type = "auto";
86  
87          if (this.type.equals(TYPE_STRING)) {
88              ext_type = "string";
89          } else if (this.type.equals(TYPE_INTEGER)) {
90              ext_type = "int";
91          } else if (this.type.equals(TYPE_DOUBLE)) {
92              ext_type = "number";
93          } else if (this.type.equals(TYPE_BOOLEAN)) {
94              ext_type = "boolean";
95          } else if (this.type.equals(TYPE_DATE)) {
96              ext_type = "date";
97          } else if (this.type.equals(TYPE_TIMESTAMP)) {
98              ext_type = "date";
99          }
100 
101         return ext_type;
102     }

138     public JSONObject toJSONObject() throws JSONException {
139         JSONObject j = new JSONObject();
140         j.put("id", id);
141         j.put("name", name);
142         j.put("alias", alias);
143         j.put("type", type);
144         return j;
145     }
146 
147     @Override
148     public int hashCode() {
149         int hash = 7;
150         hash = 97 * hash + (this.name != null ? this.name.hashCode() : 0);
151         hash = 97 * hash + (this.alias != null ? this.alias.hashCode() : 0);
152         hash = 97 * hash + (this.type != null ? this.type.hashCode() : 0);
153         return hash;
154     }
155     
156     @Override
157     public boolean equals(Object obj) {
158         if (obj == null) {
159             return false;
160         }
161         if (getClass() != obj.getClass()) {
162             return false;
163         }
164         final AttributeDescriptor other = (AttributeDescriptor) obj;
165         if ((this.name == null) ? (other.name != null) : !this.name.equals(other.name)) {
166             return false;
167         }
168         if ((this.alias == null) ? (other.alias != null) : !this.alias.equals(other.alias)) {
169             return false;
170         }
171         if ((this.type == null) ? (other.type != null) : !this.type.equals(other.type)) {
172             return false;
173         }
174         return true;
175     }
