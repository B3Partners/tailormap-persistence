308     /** To prevent a lot of SQL requests walking a tree structure of entities,
309      * load all layers using an efficient query. The Layers.children collections
310      * are not initialized, but can be reconstructed from the list of all Layers
311      * for this service returned by the query. Call Layer.getLayerChildrenCache() 
312      * to retrieve it without causing a SQL query.
313      * 
314      * The cache is not updated on changes, so will only represent the database
315      * state when loadLayerTree() was last called.
316      *
317      * @param em the entity manager to use
318      * @return the list of layers of this service
319      */
320     public List<Layer> loadLayerTree(EntityManager em) {
321         if(layers != null) {
322             return layers;
323         }
324         
325         if(!em.contains(this)) {
326             // Not a persistent entity (for example when loading user specified 
327             // service)
328             return Collections.EMPTY_LIST;
329         }
330         
331         // Retrieve layer tree structure in single query
332         layers = em.createNamedQuery("getLayerTree")
333             .setParameter("rootId", topLayer.getId())
334             .getResultList();   
335       
336         childrenByParent = new HashMap<>();
337         for(Layer l: layers) {               
338             if(l.getParent() != null) {
339                 List<Layer> parentChildren = childrenByParent.get(l.getParent());
340                 if(parentChildren == null) {
341                     parentChildren = new ArrayList<>();
342                     childrenByParent.put(l.getParent(), parentChildren);
343                 }
344                 parentChildren.add(l);
345             }
346         }      
347         return layers;
348     }
349     
350     public List<Layer> getLayerChildrenCache(Layer l, EntityManager em) {
351         if(childrenByParent != null) {
352             
353             if(!em.getEntityManagerFactory().getPersistenceUnitUtil().isLoaded(l.getChildren())) {
354                 List<Layer> childrenList = childrenByParent.get(l);
355                 if(childrenList == null) {
356                     return Collections.EMPTY_LIST;
357                 } else {
358                     return childrenList;
359                 }
360             } else {
361                 return l.getChildren();
362             }
363         } else {
364             return l.getChildren();
365         }
366     }

369     /**
370      * Gets a single layer without loading all layers. If multiple layers exist
371      * with the same name, a random non-virtual layer is returned.
372      *
373      * @param layerName the name of the layer to find
374      * @param em the EntityManager to use
375      * @return the named layer
376      */
377     public Layer getSingleLayer(final String layerName, EntityManager em) {
378         try {
379             return (Layer)em.createQuery(
380                       "from Layer where service = :service "
381                     + "and name = :n order by virtual desc")
382                     .setParameter("service", this)
383                     .setParameter("n", layerName)
384                     .setMaxResults(1)
385                     .getSingleResult();
386 
387         } catch (NoResultException nre) {
388             return null;
389         }        
390     }

392     /**
393      * Returns the layer with the given name in this server. The first layer in
394      * a depth-first tree traversal with the name is returned. If a child has
395      * the same name as its parent, the child is returned.
396      *
397      * @param layerName the layer name to search for
398      * @param em the entity manager to use
399      * @return the Layer or null if not found
400      */
401     public Layer getLayer(final String layerName, EntityManager em) {
402         loadLayerTree(em);
403         
404         if(layerName == null || topLayer == null) {
405             return null;
406         }
407         
408         final MutableObject<Layer> layer = new MutableObject<>(null);
409         
410         topLayer.accept((l, em1) -> {
411             if(StringUtils.equals(l.getName(),layerName)) {
412                 layer.setValue(l);
413                 return false;
414             }
415             return true;
416         },em);
417         
418         return layer.getValue();
419     }
