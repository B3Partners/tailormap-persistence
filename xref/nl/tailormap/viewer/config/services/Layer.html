218     public void update(Layer update, Set<String> additionalUpdatableDetails) {
219         if (!getName().equals(update.getName())) {
220             throw new IllegalArgumentException("Cannot update layer with properties from layer with different name!");
221         }
222 
223         virtual = update.virtual;
224         queryable = update.queryable;
225         filterable = update.filterable;
226         title = update.title;
227         minScale = update.minScale;
228         maxScale = update.maxScale;
229 
230         if (!boundingBoxes.equals(update.boundingBoxes)) {
231             boundingBoxes.clear();
232             boundingBoxes.putAll(update.boundingBoxes);
233         }
234         if (!crsList.equals(update.crsList)) {
235             crsList.clear();
236             crsList.addAll(update.crsList);
237         }
238         if (!keywords.equals(update.keywords)) {
239             keywords.clear();
240             keywords.addAll(update.keywords);
241         }
242 
243         // updateableDetails maps are used for clearing only details which are
244         // set by loading metadata, leave details set by other code alone
245 
246         for (String s : updatableDetails) {
247             details.remove(s);
248         }
249         if (additionalUpdatableDetails != null) {
250             for (String s : additionalUpdatableDetails) {
251                 details.remove(s);
252             }
253         }
254         // update all metadata loaded details
255         details.putAll(update.getDetails());
256         RemoveEmptyMapValuesUtil.removeEmptyMapValues(details);
257 
258         legendImageUrl = update.legendImageUrl;
259 
260         // tileSet ignored -- only for tile services!
261     }

263     /**
264      * Copy user modified properties of given layer onto this instance. Used for
265      * updating the topLayer. Not called for other layers, those instances are
266      * updated with update().
267      *
268      * @param other the source of properties to copy
269      */
270     public void copyUserModifiedProperties(Layer other) {
271         setTitleAlias(other.getTitleAlias());
272         getReaders().clear();
273         getReaders().addAll(other.getReaders());
274         getWriters().clear();
275         getWriters().addAll(other.getWriters());
276     }

289     public interface Visitor {
290         boolean visit(Layer l, EntityManager em);
291     }

293     /**
294      * Do a depth-first traversal while the visitor returns true. Uses the call
295      * stack to save layers yet to visit.
296      *
297      * @param visitor the Layer.Visitor
298      * @param em      the entity manager to use
299      * @return {@code true} if visitor accepted all layers
300      */
301     public boolean accept(Layer.Visitor visitor, EntityManager em) {
302         for (Layer child : getCachedChildren(em)) {
303             if (!child.accept(visitor, em)) {
304                 return false;
305             }
306         }
307         return visitor.visit(this, em);
308     }

320     public JSONObject toJSONObject() throws JSONException {
321         JSONObject o = new JSONObject();
322 
323         o.put("id", id);
324         o.put("serviceId", service.getId());
325         o.put("name", name);
326 
327         o.put("virtual", virtual);
328         o.put("queryable", queryable);
329         o.put("filterable", filterable);
330 
331         o.put("userlayer", userlayer);
332 
333         if (title != null) {
334             o.put("title", title);
335         }
336         if (titleAlias != null) {
337             o.put("titleAlias", titleAlias);
338         }
339         if (legendImageUrl != null) {
340             o.put("legendImageUrl", legendImageUrl);
341         }
342         if (minScale != null) {
343             if (minScale.isNaN() || minScale.isInfinite()) {
344                 log.error("Can't use minScale: " + minScale + " of Servicelayer" + this.service.getName() + " - " + this.name);
345             } else {
346                 o.put("minScale", minScale);
347             }
348         }
349         if (maxScale != null) {
350             if (maxScale.isNaN() || maxScale.isInfinite()) {
351                 log.error("Can't use maxScale: " + maxScale + " of Servicelayer" + this.service.getName() + " - " + this.name);
352             } else {
353                 o.put("maxScale", maxScale);
354             }
355         }
356 
357         o.put("hasFeatureType", featureType != null);
358         if (featureType != null) {
359             o.put("featureTypeName", featureType.getTypeName());
360             o.put("featureTypeId", featureType.getId());
361         }
362 
363         /* Only include "interesting" details in JSON */
364 
365         if (!details.isEmpty()) {
366             JSONObject d = new JSONObject();
367             o.put("details", d);
368             for (Map.Entry<String, ClobElement> e : details.entrySet()) {
369                 if (interestingDetails.contains(e.getKey())) {
370                     d.put(e.getKey(), e.getValue().getValue());
371                 }
372             }
373         }
374 
375         if (tileset != null) {
376             o.put("tileHeight", tileset.getHeight());
377             o.put("tileWidth", tileset.getWidth());
378             if (tileset.getResolutions() != null) {
379                 StringBuilder resolutions = new StringBuilder();
380                 for (Double d : tileset.getResolutions()) {
381                     if (resolutions.length() > 0) {
382                         resolutions.append(",");
383                     }
384                     resolutions.append(d.toString());
385                 }
386                 o.put("resolutions", resolutions.toString());
387             }
388         }
389 
390         if (boundingBoxes.size() == 1) {
391             BoundingBox bbox = boundingBoxes.values().iterator().next();
392             o.put("bbox", bbox.toJSONObject());
393         }
394         JSONArray sets = new JSONArray();
395         o.put("matrixSets", sets);
396         for (TileMatrixSet matrixSet : matrixSets) {
397             sets.put(matrixSet.toJSONObject());
398         }
399 
400         return o;
401     }
