24  import javax.persistence.Entity;
25  import javax.persistence.EntityManager;
26  import javax.persistence.FetchType;
27  import javax.persistence.OneToMany;
28  import java.util.ArrayList;
29  import java.util.List;
30  
31  /**
32   *
33   * @author Matthijs Laan
34   * @author Meine Toonen
35   */
36  @Entity
37  @DiscriminatorValue(TileService.PROTOCOL)
38  public class TileService extends GeoService {
39      private static final Log log = LogFactory.getLog(TileService.class);
40      public static final String PROTOCOL = "tiled";
41      public static final String PARAM_RESOLUTIONS = "resolutions";
42      public static final String PARAM_TILESIZE = "tileSize";
43      public static final String PARAM_TILINGPROTOCOL = "tilingProtocol";
44      public static final String PARAM_SERVICENAME = "ServiceName";
45      public static final String PARAM_SERVICEBBOX= "serviceBbox";
46      public static final String PARAM_IMAGEEXTENSION= "imageExtension";
47      public static final String PARAM_CRS= "crs";
48      
49      public static final String TILING_PROTOCOL_WMTS = "WMTS";
50      public static final String TILING_PROTOCOL_TMS = "TMS";
51      
52      @OneToMany(cascade=CascadeType.ALL, fetch=FetchType.LAZY, mappedBy="tileService")
53      private List<TileMatrixSet> matrixSets = new ArrayList<>();
54  
55  
56      private String tilingProtocol;
57  
58      /**
59       * Get the layer that contains the tiling settings etc.
60       * @return the layer with tiling settings
61       */
62      public Layer getTilingLayer(){
63          if (this.getTopLayer()!=null && this.getTopLayer().getChildren().size()>0){
64              return this.getTopLayer().getChildren().get(0);
65          }
66          return null;
67      }
68      
69  
70      
71      // <editor-fold desc="Getters and setters"  defaultstate="collapsed">
72      
73      public List<TileMatrixSet> getMatrixSets() {
74          return matrixSets;
75      }
76  
77      public void setMatrixSets(List<TileMatrixSet> matrixSets) {
78          this.matrixSets = matrixSets;
79      }
80      
81      public String getTilingProtocol() {
82          return tilingProtocol;
83      }
84  
85      public void setTilingProtocol(String tilingProtocol) {
86          this.tilingProtocol = tilingProtocol;
87      }
88  
89      //@Override
90      public void checkOnline(EntityManager em) throws Exception {
91      }
92      
93      // </editor-fold>
94  }
