17  package nl.tailormap.viewer.config.services;
18  
19  import javax.persistence.CollectionTable;
20  import javax.persistence.Column;
21  import javax.persistence.ElementCollection;
22  import javax.persistence.Entity;
23  import javax.persistence.Id;
24  import javax.persistence.JoinColumn;
25  import javax.persistence.OrderColumn;
26  import java.util.ArrayList;
27  import java.util.List;
28  
29  /**
30   *
31   * @author Matthijs Laan
32   */
33  @Entity
34  public class TileSet {
35  
36      @Id
37      private String name;
38  
39      private int width;
40      private int height;
41  
42      @ElementCollection
43      @OrderColumn(name="list_index")
44      @CollectionTable(
45              joinColumns = @JoinColumn(name = "tile_set")
46      )
47      @Column(name="resolution")
48      private List<Double> resolutions = new ArrayList<Double>();
49  
50      public int getHeight() {
51          return height;
52      }
53  
54      public void setHeight(int height) {
55          this.height = height;
56      }
57  
58      public String getName() {
59          return name;
60      }
61  
62      public void setName(String name) {
63          this.name = name;
64      }
65  
66      public List<Double> getResolutions() {
67          return resolutions;
68      }
69  
70      public void setResolutions(List<Double> resolutions) {
71          this.resolutions = resolutions;
72      }
73      
74      public void setResolutions(String res){
75          this.resolutions = new ArrayList<Double>();
76          String[] resTokens = res.split(",");
77          for (int i = 0; i < resTokens.length; i++){
78              this.resolutions.add(Double.parseDouble(resTokens[i].trim()));
79          }
80      }
81  
82      public int getWidth() {
83          return width;
84      }
85  
86      public void setWidth(int width) {
87          this.width = width;
88      }
89  }
