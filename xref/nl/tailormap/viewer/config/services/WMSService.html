17  package nl.tailormap.viewer.config.services;
18  
19  import org.apache.commons.logging.Log;
20  import org.apache.commons.logging.LogFactory;
21  
22  import javax.persistence.Column;
23  import javax.persistence.DiscriminatorValue;
24  import javax.persistence.Entity;
25  import javax.persistence.EnumType;
26  import javax.persistence.Enumerated;
27  
28  /**
29   * Entity for saving WMS service metadata. Enables the administration module to
30   * easily work with WMS service entities and the viewer to quickly marshall the
31   * metadata without having to do a GetCapabilities request each time the viewer 
32   * starts.
33   * <p>
34   * This requires an option to update the metadata should the service change, so
35   * this class implements Updatable.
36   * <p>
37   * @author Matthijs Laan
38   */
39  @Entity
40  @DiscriminatorValue(WMSService.PROTOCOL)
41  public class WMSService extends GeoService {
42      private static final Log log = LogFactory.getLog(WMSService.class);
43      
44      /**
45       * JPA DiscriminatorValue for this class.
46       */
47      public static final String PROTOCOL = "wms";
48  
49      /**
50       * Parameter to specify the value for #getOverrideUrl().
51       */
52      public static final String PARAM_OVERRIDE_URL = "overrideUrl";
53  
54      /**
55       * Parameter to specify the value for {@code skipDiscoverWFS}.
56       *
57       * @see #skipDiscoverWFS
58       */
59      public static final String PARAM_SKIP_DISCOVER_WFS = "skipDiscoverWFS";
60  
61     
62      /* Detail key under which "true" is saved in details if in the WMS capabilities 
63       * the <UserDefinedSymbolization> element has a positive SupportSLD attribute.
64       */ 
65      public static final String DETAIL_SUPPORT_SLD = "SupportSLD";
66      
67      /* Detail key under which "true" is saved in details if in the WMS capabilities 
68       * the <UserDefinedSymbolization> element has a positive UserStyle attribute.
69       */ 
70      public static final String DETAIL_USER_STYLE = "UserStyle";
71      
72  
73      /**
74       * Additional persistent property for this subclass, so type must be nullable.
75       */
76      private Boolean overrideUrl;
77  
78      /**
79       * Additional persistent property for this subclass to remember wether to
80       * search for and join WFS attribute sources, so type must be nullable.
81       */
82      @Column(name="skip_discoverwfs")
83      private Boolean skipDiscoverWFS;
84      
85      @Enumerated(EnumType.STRING)
86      private WMSExceptionType exception_type = WMSExceptionType.Inimage;
87  
88      /**
89       * Whether to use the original URL the Capabilities was loaded with or the
90       * URL the WMS reports it is at in the Capabilities. Sometimes the URL reported
91       * by the WMS is outdated, but it can also be used to migrate the service
92       * to another URL or load Capabilities from a static XML Capabilities document
93       * on a simple HTTP server. According to the standard the URL in the Capabilities
94       * should be used, so set this to false by default except if the user requests
95       * an override.
96       *
97       * @return true when set
98       */
99      public Boolean getOverrideUrl() {
100         return overrideUrl;
101     }
102 
103     public void setOverrideUrl(Boolean overrideUrl) {
104         this.overrideUrl = overrideUrl;
105     }
106     
107     public WMSExceptionType getException_type() {
108         return exception_type;
109     }
110 
111     public void setException_type(WMSExceptionType exception_type) {
112         this.exception_type = exception_type;
113     }
114 
115     public Boolean getSkipDiscoverWFS() {
116         return skipDiscoverWFS;
117     }
118 
119     public void setSkipDiscoverWFS(Boolean skipDiscoverWFS) {
120         this.skipDiscoverWFS = skipDiscoverWFS;
121     }
122 
123     @Override
124     public String toString() {
125         return String.format("WMS service \"%s\" at %s", getName(), getUrl());
126     }
127 
128     //</editor-fold>
129 
130 }
